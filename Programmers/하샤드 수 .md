### 하샤드 수

- 문제 : [12947](https://programmers.co.kr/learn/courses/30/lessons/12947)

~~~javascript
function solution(x) {
    let sum = 0;
    const stringA = String(x).split("");

    for(let i = 0; i < stringA.length; i++){
        sum += parseInt(stringA[i]);
    }
    return (x % sum == 0) ? true : false;
 }

~~~
