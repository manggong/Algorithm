### 서울에서 김서방 찾기

- 문제 : [12919](https://programmers.co.kr/learn/courses/30/lessons/12919)

~~~java
class Solution {
  public String solution(String[] seoul) {
      String answer= "";
      String kim = "Kim";
      for(int i=0; i<seoul.length; i++){
          if(seoul[i].equals(kim)){
              answer= "김서방은 "+i+"에 있다";
          }
      }
      return answer;
  }
}
~~~
