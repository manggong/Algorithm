### 제일 작은 수 제거하기

- 문제 : [12935](https://programmers.co.kr/learn/courses/30/lessons/12935)

~~~javascript
function solution(arr) {
    var answer = [];
    if(arr.length == 1){
        answer[0] = -1
    } else {
        const min = Math.min.apply(null, arr)
        const idx = arr.indexOf(min)
        arr.splice(idx,1)
        answer = arr
    }
    return answer;
}

~~~
