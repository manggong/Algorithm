### 나누어 떨어지는 숫자 배열

- 문제 : [12910](https://programmers.co.kr/learn/courses/30/lessons/12910)

~~~javascript
function solution(arr, divisor) {
  let answer = [];
  let count = 0;

  for (let i = 0; i < arr.length; i++) {
    if (arr[i] % divisor == 0) {
      answer.push(arr[i]);
      count++;
    }
  }
  if (count == 0) {
    answer = [-1];
  }
  answer.sort(function(a, b) {
    return a - b;
  });
  return answer;
}
~~~
